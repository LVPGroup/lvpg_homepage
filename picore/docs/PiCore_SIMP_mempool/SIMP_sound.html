lemma Unprecond_sound:
  assumes p0: " ⊨⇩I P sat⇩p [pre, rely, guar, post]"
    and  p1: " ⊨⇩I P sat⇩p [pre', rely, guar, post]"
   shows " ⊨⇩I P sat⇩p [pre ∪ pre', rely, guar, post]"
proof -
{
  fix s c
  assume "c ∈ cpts_of_p P s ∩ assume_p(pre ∪ pre', rely)"
  hence a1: "c ∈ cpts_of_p P s" and
        a2: "c ∈ assume_p(pre ∪ pre', rely)" by auto
  hence "c ∈ assume_p(pre, rely) ∨ c ∈ assume_p(pre', rely)"
    by (metis (no_types, lifting) CollectD CollectI Un_iff assume_p_def prod.simps(2))
  hence "c ∈ commit_p(guar, post)"
    proof
      assume "c ∈ assume_p (pre, rely)"
      with p0 a1 show "c ∈ commit_p (guar, post)" 
        unfolding prog_validity_def by auto
    next
      assume "c ∈ assume_p (pre', rely)"
      with p1 a1 show "c ∈ commit_p (guar, post)" 
        unfolding prog_validity_def by auto
    qed
}
then show ?thesis unfolding prog_validity_def by auto
qed

  with a0 a2 have "c ∈ commit_p(guar, post)" by auto

qed

subsubsection‹Soundness of None rule›

lemma none_all_none: "c!0=(None,s) ∧ c ∈ cpts_p ⟹ ∀i<length c. fst (c ! i) = None"
proof(induct c arbitrary:s)
  case Nil
  then show ?case by simp
next
  case (Cons a c)
  assume p1: "⋀s. c ! 0 = (None, s) ∧ c ∈ cpts_p ⟹ ∀i<length c. fst (c ! i) = None"
    and  p2: "(a # c) ! 0 = (None, s) ∧ a # c ∈ cpts_p"
  hence a0: "a = (None,s)" by simp
  thus ?case
    proof(cases "c = []")
      case True
      with a0 show ?thesis by auto
    next
      case False
      assume b0: "c ≠ []"
      with p2 have c_cpts: "c ∈ cpts_p" using tl_in_cptn by fast
      from b0 obtain c' and b where bc': "c = b # c'"
        using list.exhaust by blast 
      from a0 have "¬ a -impc→ b" by(force elim: ptran.cases)
      with p2 have "a -imppe→ b" using bc' etran_or_ctran2_disjI3[of "a#c" 0] by auto
      hence "fst b = None" using petran.cases
        by (metis a0 prod.collapse) 
      with p1 bc' c_cpts have "∀i<length c. fst (c ! i) = None"
        by (metis nth_Cons_0 prod.collapse)
      with a0 show ?thesis
        by (simp add: nth_Cons') 
    qed
      
qed

lemma None_sound_h: "∀x. x ∈ pre ⟶ (∀y. (x, y) ∈ rely ⟶ y ∈ pre) ⟹
         pre ⊆ post ⟹
         snd (c ! 0) ∈ pre ⟹
         c ≠ [] ⟹ ∀i. Suc i < length c ⟶ (snd (c ! i), snd (c ! Suc i)) ∈ rely 
      ⟹ i < length c ⟹ snd (c ! i) ∈ pre"
apply(induct i) by auto

lemma None_sound:
  "⟦ stable pre rely; pre ⊆ post ⟧
  ⟹ ⊨⇩I None sat⇩p [pre, rely, guar, post]"
proof -
  assume p0: "stable pre rely"
    and  p2: "pre ⊆ post"
  {
    fix s c
    assume a0: "c ∈ cpts_of_p None s ∩ assume_p(pre, rely)"
    hence c1: "c!0=(None,s) ∧ c ∈ cpts_p" by (simp add:cpts_of_p_def)
    from a0 have c2: "gets_p (c!0) ∈ pre ∧ (∀i. Suc i<length c ⟶ 
               c!i -imppe→ c!(Suc i) ⟶ (gets_p (c!i), gets_p (c!Suc i)) ∈ rely)"
      by (simp add:assume_p_def)
    from c1 have c_ne_empty: "c ≠ []"
      by auto 
    from c1 have c_all_none: "∀i<length c. fst (c ! i) = None" using none_all_none by fast
   
    (* have "(∀i. Suc i<length c ⟶ 
               c!i -impc→ c!(Suc i) ⟶ (gets_p (c!i), gets_p (c!Suc i)) ∈ guar)" *)
    {
      fix i
      assume suci: "Suc i<length c"
        and cc: "c!i -impc→ c!(Suc i)"
      from suci c_all_none have "c!i -imppe→ c!(Suc i)"
        by (metis Suc_lessD petran.intros prod.collapse) 
      with cc have"(gets_p (c!i), gets_p (c!Suc i)) ∈ guar"
        using c1 etran_or_ctran2_disjI1 suci by auto
    } 
    moreover
    {
      assume last_none: "getspc_p (last c) = None"
      from c2 c_all_none have "∀i. Suc i<length c ⟶ (gets_p (c!i), gets_p (c!Suc i)) ∈ rely"
        by (metis Suc_lessD petran.intros prod.collapse) 
      with p0 p2 c2 c_ne_empty have "∀i. i < length c ⟶ snd (c ! i) ∈ pre" 
        apply(simp add:gets_p_def stable_def) apply clarify using None_sound_h by blast
      with p2 c_ne_empty have "gets_p (last c) ∈ post" apply(simp add:gets_p_def)
        using One_nat_def c_ne_empty last_conv_nth by force 
          
    }
    ultimately have "c ∈ commit_p(guar, post)" by (simp add:commit_p_def)
  }
  thus "⊨⇩I None sat⇩p [pre, rely, guar, post]" using prog_validity_def by blast
qed
