section ‹Syntax of SIMP language›

theory SIMP_lang 
imports Main
begin

type_synonym 's bexp = "'s set"

datatype 's prog =
    Basic "'s ⇒'s"
  | Seq "'s prog" "'s prog"
  | Cond "'s bexp" "'s prog" "'s prog"
  | While "'s bexp" "'s prog"
  | Await "'s bexp" "'s prog"
  | Nondt "('s × 's) set"

end
