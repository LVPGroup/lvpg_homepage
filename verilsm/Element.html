         let 
            attrs' = int(of_char attrs);
            mode = if ((attrs' AND ATTR_RO) ≠ 0) ∧ ¬(((attrs' AND ATTR_DIR) ≠ 0 )∧ ¬(rodir( sbi_options sbi)) ≠ 0)
                    then ((int m) AND (NOT S_IWUGO) )
                    else (int m);
           ret = (if (attrs' AND ATTR_DIR) ≠ 0 then
                     bitOR (mode AND (NOT (int(fs_dmask(sbi_options sbi))))) S_IFDIR
                 else
                    bitOR (mode AND (NOT (int(fs_fmask(sbi_options sbi))))) S_IFREG ) 
         in (nat ret)
        "
